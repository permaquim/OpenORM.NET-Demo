CREATE VIEW dbo.VistaUsuario
AS
SELECT        dbo.Usuario.Nombre, dbo.Usuario.Apellido, dbo.Usuario.Edad, dbo.Pais.Nombre AS NombrePais, dbo.Pais.ContinenteId, dbo.Continente.Nombre AS NombreContinente
FROM            dbo.Continente INNER JOIN
                         dbo.Pais ON dbo.Continente.Id = dbo.Pais.ContinenteId INNER JOIN
                         dbo.Usuario ON dbo.Pais.Id = dbo.Usuario.PaisId
